// Translate App
function applyLanguage(lang) {
  currentLanguage = lang;
  localStorage.setItem('lang', lang);

  const appWrapper = document.getElementById('app-wrapper');
  const langText = document.getElementById('lang-text');

  if (lang === 'ar') {
    appWrapper.className = 'app-container rtl';
    langText.innerText = 'Français';
  } else {
    appWrapper.className = 'app-container ltr';
    langText.innerText = 'العربية';
  }

  // Update DOM translation nodes
  document.querySelectorAll('[data-t]').forEach(element => {
    const key = element.getAttribute('data-t');
    if (translations[lang][key]) {
      element.innerText = translations[lang][key];
    }
  });

  // Update input placeholders
  document.querySelectorAll('[data-t-placeholder]').forEach(element => {
    const key = element.getAttribute('data-t-placeholder');
    if (translations[lang][key]) {
      element.setAttribute('placeholder', translations[lang][key]);
    }
  });

  // Recreate Lucide Icons
  lucide.createIcons();

  // Redraw charts because label alignments and legends change layout
  if (earningsChartInstance && statusChartInstance) {
    updateChartsLocalization();
  }
}

// Switch Active View Tab
async function switchTab(tabName) {
  // Update link styling
  document.querySelectorAll('.menu-link').forEach(link => {
    if (link.getAttribute('data-tab') === tabName) {
      link.classList.add('active');
    } else {
      link.classList.remove('active');
    }
  });

  // Update view panel visibility
  document.querySelectorAll('.view-panel').forEach(panel => {
    if (panel.getAttribute('id') === 'view-' + tabName) {
      panel.classList.add('active');
    } else {
      panel.classList.remove('active');
    }
  });

  // Update Page Title
  document.getElementById('view-title').innerText = translations[currentLanguage]['title_' + tabName] || translations[currentLanguage]['tab_' + tabName];

  // Refresh active tab data
  await loadAllData();

  if (tabName === 'dashboard') {
    refreshDashboard();
  } else if (tabName === 'customers') {
    refreshCustomers();
  } else if (tabName === 'parcels') {
    refreshParcelsForm();
  } else if (tabName === 'payments') {
    refreshPayments();
  } else if (tabName === 'reports') {
    refreshReports();
  }

  // Auto trigger icons
  lucide.createIcons();
}

// Calculate Parcel Cost based on dimensions and weight (volumetric weight calculation)
function calculateParcelCost() {
  const weight = parseFloat(document.getElementById('parcel-weight').value) || 0;
  const len = parseFloat(document.getElementById('parcel-length').value) || 0;
  const w = parseFloat(document.getElementById('parcel-width').value) || 0;
  const h = parseFloat(document.getElementById('parcel-height').value) || 0;
  const destination = document.getElementById('parcel-destination').value;

  // Volumetric weight formula: (L * W * H) / 5000
  const volumetricWeight = (len * w * h) / 5000;
  const chargeableWeight = Math.max(weight, volumetricWeight);

  // Rate factor based on destination
  let baseRate = 12.0; // Default domestic rate factor
  if (destination === 'فرنسا' || destination === 'ألمانيا' || destination === 'بلجيكا' || destination === 'المملكة المتحدة') {
    baseRate = 25.0; // Europe rate
  } else if (destination === 'الولايات المتحدة') {
    baseRate = 35.0; // US rate
  } else if (destination === 'تركيا') {
    baseRate = 18.0; // Turkey rate
  } else if (destination !== 'المملكة العربية السعودية') {
    baseRate = 15.0; // Middle East / other international rate
  }

  const cost = chargeableWeight * baseRate;
  document.getElementById('parcel-cost').value = cost.toFixed(2);

  // Also update payment amount values depending on payment status selection
  const status = document.getElementById('parcel-payment-status').value;
  const amountPaidInput = document.getElementById('parcel-amount-paid');

  if (status === 'مدفوعة') {
    amountPaidInput.value = cost.toFixed(2);
  } else if (status === 'غير مدفوعة') {
    amountPaidInput.value = 0.00;
  }
}

// Handle Customer Form Submit
async function handleCustomerSubmit(e) {
  e.preventDefault();
  const id = document.getElementById('customer-id').value;
  const name = document.getElementById('customer-name').value.trim();
  const phone = document.getElementById('customer-phone').value.trim();
  const email = document.getElementById('customer-email').value.trim();
  const address = document.getElementById('customer-address').value.trim();

  const customerData = { name, phone, email, address };

  try {
    if (id) {
      // Update
      await window.api.updateCustomer(id, customerData);
    } else {
      // Add
      await window.api.addCustomer(customerData);
    }

    closeAllModals();
    showToast(translations[currentLanguage].msg_customer_saved, 'success');
    
    // Refresh
    const activeLink = document.querySelector('.menu-link.active').getAttribute('data-tab');
    if (activeLink === 'customers') {
      refreshCustomers();
    } else if (activeLink === 'parcels') {
      refreshParcelsForm();
    }
  } catch (err) {
    console.error(err);
    showToast('فشل حفظ العميل / Échec de sauvegarde', 'error');
  }
}

// Simulate Document/Image upload dropzone
function simulateAttachmentUpload() {
  // Creating virtual input
  const input = document.createElement('input');
  input.type = 'file';
  input.accept = 'image/*,application/pdf';
  
  input.onchange = (e) => {
    const file = e.target.files[0];
    if (file) {
      if (file.size > 5 * 1024 * 1024) {
        showToast('الملف كبير جداً (أقصى حد 5 ميغابايت) / Fichier trop volumineux', 'error');
        return;
      }
      
      const reader = new FileReader();
      reader.onload = () => {
        uploadedFileBase64 = reader.result;
        
        // Show Preview
        const previewContainer = document.getElementById('upload-preview-container');
        previewContainer.style.display = 'block';
        
        if (file.type.startsWith('image/')) {
          previewContainer.innerHTML = `
            <div class="upload-preview">
              <img src="${reader.result}" alt="Preview">
              <div>
                <p style="font-size:12px; font-weight:700;">${file.name}</p>
                <span style="font-size:10px; color:var(--text-muted);">${(file.size/1024).toFixed(1)} KB</span>
              </div>
            </div>
          `;
        } else {
          previewContainer.innerHTML = `
            <div class="upload-preview">
              <i data-lucide="file-text" style="font-size: 32px; color: var(--color-primary);"></i>
              <div>
                <p style="font-size:12px; font-weight:700;">${file.name}</p>
                <span style="font-size:10px; color:var(--text-muted);">${(file.size/1024).toFixed(1)} KB</span>
              </div>
            </div>
          `;
          lucide.createIcons();
        }
      };
      reader.readAsDataURL(file);
    }
  };
  
  input.click();
}

// Helper to match status string to translations key
function getStatusKey(status) {
  if (status === 'قيد الاستلام' || status === 'Received' || status === 'status_received') return 'received';
  if (status === 'قيد المعالجة' || status === 'Processing' || status === 'status_processing') return 'processing';
  if (status === 'في الطريق' || status === 'In Transit' || status === 'status_intransit') return 'intransit';
  if (status === 'وصلت إلى بلد الوجهة' || status === 'Arrived' || status === 'status_arrived') return 'arrived';
  if (status === 'تم التسليم' || status === 'Delivered' || status === 'status_delivered') return 'delivered';
  if (status === 'ملغاة أو مرفوضة' || status === 'Cancelled' || status === 'status_cancelled') return 'cancelled';
  return 'received';
}

// Render vertical tracking timeline
function renderTrackingTimeline(parcel) {
  const timelineItems = document.getElementById('tracking-timeline-items');
  timelineItems.innerHTML = '';

  const statuses = [
    { key: 'received', title: translations[currentLanguage].status_received, desc: currentLanguage === 'ar' ? 'تم استلام الشحنة وتجهيزها في مقر الوكالة' : 'Le colis a été reçu et préparé à l\'agence' },
    { key: 'processing', title: translations[currentLanguage].status_processing, desc: currentLanguage === 'ar' ? 'الشحنة تخضع للتغليف والتدقيق الأمني' : 'Le colis est en cours d\'emballage et d\'inspection' },
    { key: 'intransit', title: translations[currentLanguage].status_intransit, desc: currentLanguage === 'ar' ? 'غادرت الشحنة الوكالة وهي قيد النقل الدولي' : 'Le colis a quitté l\'agence, en transit international' },
    { key: 'arrived', title: translations[currentLanguage].status_arrived, desc: currentLanguage === 'ar' ? 'وصلت الشحنة إلى مركز الجمارك بالبلد المستقبل' : 'Le colis est arrivé au centre douanier du pays destinataire' },
    { key: 'delivered', title: translations[currentLanguage].status_delivered, desc: currentLanguage === 'ar' ? 'تم تسليم الشحنة للمستقبل بنجاح وتوقيع الفاتورة' : 'Le colis a été livré avec succès et signé' }
  ];

  const currentStatusKey = getStatusKey(parcel.status);
  let activeIndex = statuses.findIndex(s => s.key === currentStatusKey);
  
  if (parcel.status === 'ملغاة أو مرفوضة' || currentStatusKey === 'cancelled') {
    // Inject special rejected status timeline
    statuses.push({ key: 'cancelled', title: translations[currentLanguage].status_cancelled, desc: currentLanguage === 'ar' ? 'تم إلغاء الشحنة أو رفض استلامها من المستلم' : 'Le colis a été annulé ou refusé par le destinataire' });
    activeIndex = statuses.length - 1;
  }

  statuses.forEach((s, idx) => {
    const isActive = idx <= activeIndex;
    const isCurrent = idx === activeIndex;

    const li = document.createElement('li');
    li.className = `timeline-item ${isActive ? 'active' : ''}`;
    
    // Add date for active items
    let dateStr = '';
    if (isCurrent) {
      dateStr = new Date(parcel.updatedAt).toLocaleString(currentLanguage === 'ar' ? 'ar-EG' : 'fr-FR');

// Record Payment Submit Form
async function handleRecordPaymentSubmit(e) {
  e.preventDefault();
  const id = document.getElementById('payment-id').value;
  const amountPaid = parseFloat(document.getElementById('payment-amount-paid').value) || 0;
  const paymentMethod = document.getElementById('payment-method-select').value;
  const totalCost = parseFloat(document.getElementById('payment-total-cost').value) || 0;

  let paymentStatus = 'غير مدفوعة';

    refreshPayments();

    console.error(err);
    showToast('فشل تعديل الدفعة / Échec du paiement', 'error');
  }
}

// Render Reports and Finances
function refreshReports() {
  const period = document.getElementById('report-period').value;
  
  // Calculate analytics
  let totalRevenue = 0;
  let reportsRows = [];

  const now = new Date();
  
  // Filter payments based on period selection
  payments.forEach(pay => {
    const payDate = new Date(pay.createdAt);
    let match = false;

    if (period === 'daily') {
      match = payDate.toDateString() === now.toDateString();
    } else if (period === 'monthly') {
      match = payDate.getMonth() === now.getMonth() && payDate.getFullYear() === now.getFullYear();
    } else if (period === 'yearly') {
      match = payDate.getFullYear() === now.getFullYear();
    }

    if (match) {
      totalRevenue += pay.amountPaid;
      reportsRows.push({
        date: payDate.toLocaleDateString(currentLanguage === 'ar' ? 'ar-EG' : 'fr-FR'),
        tracking: pay.trackingNumber,
        sender: pay.senderName,
        type: currentLanguage === 'ar' ? 'شحن طرد دولي' : 'Expédition colis',
        amount: pay.amountPaid,
        status: pay.paymentStatus
      });
    }
  });

  // Operating expenses estimation (simulated fuel, handling, transport: 35% of revenue)
  const totalExpenses = totalRevenue * 0.35;
  const netProfit = totalRevenue - totalExpenses;

  // Render cards values
  document.getElementById('report-revenue-val').innerText = `$${totalRevenue.toFixed(2)}`;
  document.getElementById('report-expenses-val').innerText = `$${totalExpenses.toFixed(2)}`;
  document.getElementById('report-net-val').innerText = `$${netProfit.toFixed(2)}`;

  // Populate report table
  const tbody = document.getElementById('reports-list-table');
  tbody.innerHTML = '';

  if (reportsRows.length === 0) {
    tbody.innerHTML = `<tr><td colspan="6" style="text-align: center; color: var(--text-muted);">لا توجد حركات مالية مسجلة لهذه الفترة</td></tr>`;
    return;
  }

  reportsRows.forEach(row => {
    let badgeClass = 'unpaid';
    if (row.status === 'مدفوعة') badgeClass = 'paid';
    else if (row.status === 'مدفوعة جزئياً') badgeClass = 'partial';

    let statusText = row.status;
    if (row.status === 'مدفوعة') statusText = translations[currentLanguage].pay_paid;
    else if (row.status === 'مدفوعة جزئياً') statusText = translations[currentLanguage].pay_partial;
    else if (row.status === 'غير مدفوعة') statusText = translations[currentLanguage].pay_unpaid;

    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td class="numbers">${row.date}</td>
      <td class="numbers">${row.tracking}</td>
      <td>${row.sender}</td>
      <td>${row.type}</td>
      <td class="numbers" style="color: #10B981; font-weight:700;">$${row.amount.toFixed(2)}</td>
      <td><span class="badge ${badgeClass}">${statusText}</span></td>
    `;
    tbody.appendChild(tr);
  });
}

// Client Side CSV Export for Excel
function exportToExcel() {
  if (payments.length === 0) {
    showToast('لا توجد بيانات لتصديرها / Aucune donnée à exporter', 'info');
    return;
  }

  let csvContent = '\uFEFF'; // UTF-8 BOM to display Arabic characters properly in Excel
  
  // Headers
  if (currentLanguage === 'ar') {
    csvContent += 'رقم التتبع,العميل المرسل,المبلغ الإجمالي,المبلغ المدفوع,الدين المتبقي,حالة الدفع,طريقة الدفع\n';
  } else {
    csvContent += 'N Tracking,Expediteur,Montant Total,Montant Paye,Dette,Statut Paiement,Mode Paiement\n';
  }

  payments.forEach(pay => {
    const debt = pay.amountTotal - pay.amountPaid;
    csvContent += `${pay.trackingNumber},${pay.senderName},${pay.amountTotal},${pay.amountPaid},${debt},${pay.paymentStatus},${pay.paymentMethod}\n`;
  });

  const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.setAttribute('href', url);
  link.setAttribute('download', `ExpressCargo_Report_${new Date().toISOString().slice(0,10)}.csv`);

  document.body.appendChild(link);
  link.click();

}

// Compute dynamic data values for charts from the actual database arrays
function updateChartsData() {
  if (!earningsChartInstance || !statusChartInstance) return;

  // 1. Calculate Status Chart breakdown
  const receivedCount = parcels.filter(p => getStatusKey(p.status) === 'received').length;
  const processingCount = parcels.filter(p => getStatusKey(p.status) === 'processing').length;
  const intransitCount = parcels.filter(p => getStatusKey(p.status) === 'intransit' || getStatusKey(p.status) === 'arrived').length;
  const deliveredCount = parcels.filter(p => getStatusKey(p.status) === 'delivered').length;
  const cancelledCount = parcels.filter(p => getStatusKey(p.status) === 'cancelled').length;

  statusChartInstance.updateSeries([receivedCount, processingCount, intransitCount, deliveredCount, cancelledCount]);

  // 2. Earnings Trend computation (last 6 months dynamically)
  const monthNames = currentLanguage === 'ar' 
    ? ['ديسمبر', 'يناير', 'فبراير', 'مارس', 'أبريل', 'مايو']
    : ['Déc', 'Jan', 'Fév', 'Mar', 'Avr', 'Mai'];
  
  // Calculate sums per month (simulated grouping based on seeded mock data months)
  // Payments indices dates mapping: Dec (2025-12), Jan (2026-01), Feb (2026-02), Mar (2026-03), Apr (2026-04), May (2026-05)
  const monthlyTotals = [0, 0, 0, 0, 0, 0];

  payments.forEach(pay => {
    const date = new Date(pay.createdAt);
    const month = date.getMonth(); // 0-11
    const year = date.getFullYear();

    if (year === 2025 && month === 11) monthlyTotals[0] += pay.amountPaid; // Dec
    else if (year === 2026) {
      if (month === 0) monthlyTotals[1] += pay.amountPaid; // Jan
      else if (month === 1) monthlyTotals[2] += pay.amountPaid; // Feb
      else if (month === 2) monthlyTotals[3] += pay.amountPaid; // Mar
      else if (month === 3) monthlyTotals[4] += pay.amountPaid; // Apr
      else if (month === 4) monthlyTotals[5] += pay.amountPaid; // May
    }
  });

  earningsChartInstance.updateOptions({
    xaxis: { categories: monthNames },
    series: [{
      name: currentLanguage === 'ar' ? 'الأرباح ($)' : 'Bénéfices ($)',
      data: monthlyTotals.map(v => Math.round(v))
    }]
  });
}
